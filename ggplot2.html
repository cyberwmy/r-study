8.3 自定义主题

last_plot() + theme(
  ## 坐标轴
  axis.line = element_line(
    size = 0,
    colour = "black"
  ),
  ## 绘图区域
  panel.background = element_rect(
    fill = "white",
    # colour = "black",
    # size = 0.5
  ),
  panel.border = element_rect(
    fill = NA,
    colour = "black",
    size = 0.5
  ),
  panel.grid = element_line(colour = NA),
  panel.ontop = F,
  ## 图例
  legend.key = element_rect(
    fill = NA,
    colour = NA,
    size = NA
  ),
  legend.key.size = unit(.25, "cm"),
  legend.title = element_blank(),
  legend.position = c("bottom"),
  # legend.key.height = unit(1, "cm"),
  legend.key.width = unit(1.2, "cm"),
  legend.text = element_text(
    face = "bold.italic"
  ),
  legend.spacing.x = unit(5, "pt"),
  legend.spacing.y = unit(5, "pt"),
  ## 图形区域
  plot.margin = margin(2, 2, 2, 2, "cm"),
  plot.background = element_rect(
    fill = "white",
    colour = NA,
    size = 1
  )
)

9.3 等高点线

data(volcano)
a <- as.data.frame.table(volcano)
v <- data.frame(
  x = as.numeric(a$Var1),
  y = as.numeric(a$Var2),
  z = a$Freq
)

vp <- ggplot(v, aes(x, y, z = z))

vp + geom_contour()
vp + geom_contour_filled()
vp + geom_raster(aes(fill = z)) +
  geom_contour() +
  scale_fill_distiller(palette = "Spectral")
vp + geom_raster(aes(fill = z), interpolate = TRUE) +
  geom_contour() +
  scale_fill_distiller(palette = "Spectral")
vp + geom_tile(aes(fill = z), colour = "grey50", alpha = 0.5) +
  geom_contour(color = "red") +
  scale_fill_distiller(palette = "Spectral")
